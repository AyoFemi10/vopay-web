## Base URL

```
https://api.vopay.com/v1
```

All requests must use HTTPS. HTTP requests are rejected.

## Authentication

All protected endpoints require:

```http
Authorization: Bearer <access_token>
```

Or for server-to-server requests:

```http
X-API-Key: <api_key>
```

## Auth Endpoints

<ApiEndpoint
  method="POST"
  path="/auth/login"
  description="Authenticate with email and password. Returns access and refresh tokens."
  auth={false}
  params={[
    { name: "email", type: "string", required: true, description: "User email address" },
    { name: "password", type: "string", required: true, description: "User password" }
  ]}
  requestExample={`{
  "email": "user@example.com",
  "password": "your-password"
}`}
  responseExample={`{
  "success": true,
  "data": {
    "accessToken": "eyJ...",
    "refreshToken": "dGhp...",
    "expiresIn": 3600
  }
}`}
/>

<ApiEndpoint
  method="POST"
  path="/auth/refresh"
  description="Refresh an expired access token using a valid refresh token."
  auth={false}
  params={[
    { name: "refreshToken", type: "string", required: true, description: "Valid refresh token from login" }
  ]}
/>

<ApiEndpoint
  method="POST"
  path="/auth/logout"
  description="Invalidate the current session and revoke the refresh token."
  auth={true}
/>

## Wallet Endpoints

<ApiEndpoint
  method="GET"
  path="/wallets/balance"
  description="Get the authenticated user's wallet balance for all currencies."
  auth={true}
  responseExample={`{
  "success": true,
  "data": {
    "NGN": { "available": 1000000, "pending": 50000 },
    "USD": { "available": 0, "pending": 0 }
  }
}`}
/>

<ApiEndpoint
  method="GET"
  path="/wallets/transactions"
  description="List wallet transactions with optional filters."
  auth={true}
  params={[
    { name: "page", type: "number", required: false, description: "Page number (default: 1)" },
    { name: "limit", type: "number", required: false, description: "Results per page (default: 20, max: 100)" },
    { name: "type", type: "string", required: false, description: "Filter by type: CREDIT, DEBIT" },
    { name: "from", type: "string", required: false, description: "Start date ISO 8601" },
    { name: "to", type: "string", required: false, description: "End date ISO 8601" }
  ]}
/>

## Transfer Endpoints

<ApiEndpoint
  method="POST"
  path="/transfers"
  description="Initiate a bank transfer or internal VOPayX transfer."
  auth={true}
  params={[
    { name: "amount", type: "number", required: true, description: "Amount in lowest denomination (e.g. kobo for NGN)" },
    { name: "currency", type: "string", required: true, description: "ISO 4217 currency code (e.g. NGN, USD)" },
    { name: "reference", type: "string", required: true, description: "Unique reference for this transfer" },
    { name: "narration", type: "string", required: false, description: "Transfer description (shown on recipient statement)" },
    { name: "recipient.accountNumber", type: "string", required: true, description: "Recipient bank account number" },
    { name: "recipient.bankCode", type: "string", required: true, description: "Recipient bank code" },
    { name: "recipient.name", type: "string", required: true, description: "Recipient account name" }
  ]}
  requestExample={`{
  "amount": 50000,
  "currency": "NGN",
  "reference": "order-123",
  "narration": "Payment for order",
  "recipient": {
    "accountNumber": "0123456789",
    "bankCode": "058",
    "name": "John Doe"
  }
}`}
  responseExample={`{
  "success": true,
  "data": {
    "id": "txn_01JXXX",
    "status": "PENDING",
    "amount": 50000,
    "currency": "NGN",
    "fee": 1050,
    "reference": "order-123"
  }
}`}
/>

<ApiEndpoint
  method="GET"
  path="/transfers/:id"
  description="Get the status and details of a specific transfer."
  auth={true}
/>
